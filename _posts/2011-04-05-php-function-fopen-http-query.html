"Content-Length: " . strlen($data) . "\r\n", 'content' => $data ), ); $context = stream_context_create($opts); $html = file_get_contents('http://www.example.com', false, $context); echo $html; ?>; {% endhighlight %} 参考： * http://cn.php.net/manual/zh/function.file-get-contents.php * http://cn.php.net/manual/en/function.stream-context-create.php * http://cn.php.net/manual/zh/wrappers.http.php
